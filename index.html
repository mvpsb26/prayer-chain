📧 Email Setup
EmailJS connects to your Gmail and sends emails directly from it. Free up to 200 emails/month.
  • Go to emailjs.com and create a free account.
  • Add an Email Service → choose Gmail → connect your Google account → copy the Service ID.
  • Create an Email Template. Use these variables:
    {{to_name}}  {{slot_time}}  {{event_name}}
    Copy the Template ID.
  • Go to Account → copy your Public Key.
  • Paste all three below and save.
Google Apps Script uses your Gmail account directly — completely free with no third-party service. Requires a one-time setup in Google.
  • Go to script.google.com and click New Project.
  • Delete any existing code and paste in the script shown below.
  • Click Deploy → New Deployment → type: Web app → Execute as: Me → Who has access: Anyone → click Deploy.
  • Authorize the app when prompted (it only needs Gmail permission).
  • Copy the Web App URL and paste it below.
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var subject = "Prayer Chain Reminder: " + data.event_name;
    var body = "Dear " + data.to_name + ",\n\n"
      + "This is a reminder of your prayer time:\n\n"
      + "Event: " + data.event_name + "\n"
      + "Your time slot: " + data.slot_time + "\n\n"
      + "Thank you for your faithful commitment to prayer.\n\n"
      + "God bless you,\n"
      + "Mary's Village Paraclete Prayer Chain";
    GmailApp.sendEmail(data.to_email, subject, body);
    if (data.admin_email && data.admin_email !== data.to_email) {
      GmailApp.sendEmail(data.admin_email, "[COPY] " + subject, body);
    }
    return ContentService.createTextOutput(
      JSON.stringify({status:"ok"})
    ).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({status:"error", message: err.toString()})
    ).setMimeType(ContentService.MimeType.JSON);
  }
}
